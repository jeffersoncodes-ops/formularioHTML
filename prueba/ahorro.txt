@echo off
title Modo Ahorro EXTREMO de Bateria

echo Activando configuraciones de bajo consumo...

:: Activar plan de ahorro de energia
powercfg /setactive SCHEME_MIN

:: Bajar brillo al minimo (puede no funcionar en todos los equipos)
powershell (Get-WmiObject -Namespace root/WMI -Class WmiMonitorBrightnessMethods).WmiSetBrightness(1,20)

:: Reducir uso del CPU (limitar estado maximo al 50%)
powercfg -setacvalueindex SCHEME_MIN SUB_PROCESSOR PROCTHROTTLEMAX 50
powercfg -setdcvalueindex SCHEME_MIN SUB_PROCESSOR PROCTHROTTLEMAX 50

:: Desactivar servicios que consumen recursos
net stop "SysMain" >nul 2>&1
net stop "DiagTrack" >nul 2>&1
net stop "WSearch" >nul 2>&1

:: Cerrar apps en segundo plano comunes
taskkill /f /im OneDrive.exe >nul 2>&1
taskkill /f /im Teams.exe >nul 2>&1
taskkill /f /im Skype.exe >nul 2>&1
taskkill /f /im Chrome.exe >nul 2>&1

:: Pausar actualizaciones de Windows
net stop wuauserv >nul 2>&1

:: Reducir tiempo de apagado de pantalla (1 minuto)
powercfg /change monitor-timeout-dc 1

:: Reducir tiempo de suspension (2 minutos)
powercfg /change standby-timeout-dc 2

echo.
echo Modo ahorro EXTREMO activado.
pause
